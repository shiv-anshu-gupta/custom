1/**
2 * @file sv_publisher_instance.cc
3 * @brief Implementation of SvPublisherInstance
4 *
5 * Each instance represents one Merging Unit. It owns its config,
6 * equations, and pre-built frame cache (internal buffer).
7 *
8 * The prebuildFrames() method is the key operation:
9 * 1. Configure the encoder with this publisher's settings
10 * 2. For each sample in one AC cycle:
11 * - Generate waveform values from equations
12 * - Encode a complete SV Ethernet frame
13 * - Store in internal buffer
14 * 3. Mark state as READY
15 *
16 * After prebuild, the SvController merges all publishers' frames
17 * into a SharedBuffer for transmission.
18 */

105int SvPublisherInstance::configure(const PublisherConfig& config)
106{
107 m_config = config;
108
109 /* Validate & clamp */
110 if (m_config.channelCount < 1) m_config.channelCount = 1;
111 if (m_config.channelCount > SV_MAX_CHANNELS) m_config.channelCount = SV_MAX_CHANNELS;
112 if (m_config.asduCount != 1 && m_config.asduCount != 4 && m_config.asduCount != 8)
113 m_config.asduCount = 1;
114 if (m_config.sampleRate == 0) m_config.sampleRate = 4000;
115 if (m_config.frequency <= 0.0) m_config.frequency = 50.0;
116
117 /* Auto-derive multicast destination MAC from APPID per IEC 61850-8-1 §C.2.
118 * Only when MAC uses the standard SV multicast prefix 01:0C:CD:04:xx:xx.
119 * If user set a completely custom MAC, leave it untouched. */
120 if (m_config.dstMAC[0] == 0x01 && m_config.dstMAC[1] == 0x0C &&
121 m_config.dstMAC[2] == 0xCD && m_config.dstMAC[3] == 0x04) {
122 m_config.dstMAC[4] = (uint8_t)((m_config.appID >> 8) & 0xFF);
123 m_config.dstMAC[5] = (uint8_t)(m_config.appID & 0xFF);
124 }
125
126 /* Update equation processor */
127 m_eqProcessor.setDefaultFrequency(m_config.frequency);
128 m_eqProcessor.setSampleRate((uint32_t)m_config.sampleRate);
129
130 m_state = CONFIGURED;
131 printf("[publisher-%u] Configured: svID=%s, appID=0x%04X, rate=%llu Hz, "
132 "freq=%.0f Hz, channels=%d, asdu=%d\n",
133 m_id, m_config.svID, m_config.appID,
134 (unsigned long long)m_config.sampleRate,
135 m_config.frequency, m_config.channelCount, m_config.asduCount);
136 return 0;
137}

161/*============================================================================
162 * Frame Pre-building (fills internal buffer)
163 *
164 * IMPORTANT: This function uses the global sv_encoder which is mutex-
165 * protected. The SvController calls prebuildFrames() sequentially for
166 * each publisher (not in parallel), so there is no contention.
167 *============================================================================*/
168
169int SvPublisherInstance::prebuildFrames()
170{
171 if (m_state < CONFIGURED) {
172 snprintf(m_errorBuf, sizeof(m_errorBuf),
173 "Publisher %u: not configured", m_id);
174 m_state = FAILED;
175 return -1;
176 }
177
178 /* IEC 61850-9-2 §7.2.3: smpCnt must count 0 to (sampleRate-1) per second.
179 * We pre-build one full second of frames so smpCnt covers the complete
180 * range. The waveform naturally repeats every AC cycle, but each frame
181 * carries a unique smpCnt value. */
182 int packetsPerSecond = (int)getPacketsPerSec();
183 if (packetsPerSecond > SV_PUB_MAX_PREBUILT_FRAMES)
184 packetsPerSecond = SV_PUB_MAX_PREBUILT_FRAMES;
185 if (packetsPerSecond < 1) packetsPerSecond = 1;
186
187 printf("[publisher-%u] Building %d frames (1 sec, smpCnt 0-%d, %d ch, %d ASDU)...\n",
188 m_id, packetsPerSecond, packetsPerSecond - 1,
189 m_config.channelCount, m_config.asduCount);
190
191 /* Allocate internal buffer */
192 if (!allocFrameCache(packetsPerSecond)) {
193 snprintf(m_errorBuf, sizeof(m_errorBuf),
194 "Publisher %u: failed to allocate %d frames (%.1f MB)",
195 m_id, packetsPerSecond,
196 (double)packetsPerSecond * SV_MAX_FRAME_SIZE / (1024.0 * 1024.0));
197 m_state = FAILED;
198 return -1;
199 }
200
201 /* Configure encoder for THIS publisher's settings */
202 SvEncoderConfig enc = {};
203 strncpy(enc.svID, m_config.svID, sizeof(enc.svID) - 1);
204 enc.appID = m_config.appID;
205 enc.confRev = m_config.confRev;
206 enc.smpSynch = m_config.smpSynch;
207 memcpy(enc.srcMAC, m_config.srcMAC, 6);
208 memcpy(enc.dstMAC, m_config.dstMAC, 6);
209 enc.vlanPriority = m_config.vlanPriority;
210 enc.vlanID = m_config.vlanID;
211 enc.asduCount = m_config.asduCount;
212 enc.channelCount = m_config.channelCount;
213 sv_encoder_set_config(&enc);
214
215 /* Build one frame per sample for the full second.
216 * smpCnt = i ranges 0 to (packetsPerSecond - 1), e.g. 0-3999 at 4000 Hz.
217 * Waveform repeats every AC cycle but smpCnt is unique per frame. */
218 m_frameCount = packetsPerSecond;
219 for (int i = 0; i < packetsPerSecond; i++) {
220 double t = (double)i / (double)m_config.sampleRate;
221 int32_t samples[SV_MAX_CHANNELS] = {0};
222
223 m_eqProcessor.generate9_2LESamples(t, samples, m_config.channelCount);
224
225 size_t size = SV_MAX_FRAME_SIZE;
226 int ret = sv_encoder_encode_packet(
227 (uint32_t)i, samples, m_frames[i], &size);
228
229 if (ret != 0) {
230 snprintf(m_errorBuf, sizeof(m_errorBuf),
231 "Publisher %u: encode failed at sample %d", m_id, i);
232 m_state = FAILED;
233 return -1;
234 }
235 m_frameLens[i] = size;
236 }
237
238 m_state = READY;
239 printf("[publisher-%u] READY: %d frames, %zu bytes/frame\n",
240 m_id, m_frameCount, m_frameLens[0]);
241 return 0;
242}

248bool SvPublisherInstance::allocFrameCache(int count)
249{
250 freeFrameCache();
251
252 m_frames = new (std::nothrow) uint8_t*[count];
253 if (!m_frames) return false;
254
255 m_frameLens = new (std::nothrow) size_t[count];
256 if (!m_frameLens) {
257 delete[] m_frames; m_frames = nullptr;
258 return false;
259 }
260
261 for (int i = 0; i < count; i++) {
262 m_frames[i] = new (std::nothrow) uint8_t[SV_MAX_FRAME_SIZE];
263 if (!m_frames[i]) {
264 for (int j = 0; j < i; j++) delete[] m_frames[j];
265 delete[] m_frames; m_frames = nullptr;
266 delete[] m_frameLens; m_frameLens = nullptr;
267 return false;
268 }
269 }
270
271 m_frameCapacity = count;
272 return true;
273}
