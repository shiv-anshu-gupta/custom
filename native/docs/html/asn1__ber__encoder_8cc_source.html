1/**
2 * @file asn1_ber_encoder.cc
3 * @brief ASN.1 BER TLV Encoder for IEC 61850 Sampled Values
4 *
5 * @section ber_overview Overview
6 *
7 * This module implements ASN.1 BER (Basic Encoding Rules) encoding,
8 * specifically for IEC 61850-9-2 Sampled Values protocol. BER is a
9 * Tag-Length-Value (TLV) encoding system used to serialize structured data.
10 *
11 * @section ber_what What is ASN.1 BER?
12 *
13 * ASN.1 (Abstract Syntax Notation One) is a standard for describing data
14 * structures. BER is one encoding method for serializing ASN.1 data.
15 *
16 * Every piece of data is encoded as:
17 * ```
18 * ┌─────────┬─────────┬──────────────────────┐
19 * │ TAG │ LENGTH │ VALUE │
20 * │ (1 byte)│ (1-5 B) │ (variable) │
21 * └─────────┴─────────┴──────────────────────┘
22 * ```
23 *
24 * @section ber_example Example Encoding
25 *
26 * Encoding the string "MU001" with tag 0x80 (svID):
27 * ```
28 * Tag: 0x80
29 * Length: 0x05 (5 characters)
30 * Value: 0x4D 0x55 0x30 0x30 0x31 ('M','U','0','0','1')
31 *
32 * Result: 80 05 4D 55 30 30 31
33 * ```
34 *
35 * @section ber_sv_structure SV Message Structure
36 *
37 * @dot
38 * digraph SVStructure {
39 * rankdir=TB;
40 * node [shape=record, style=filled, fillcolor="#f0f0f0"];
41 *
42 * SAVPDU [label="{SAVPDU (0x60)|Contains entire SV message}", fillcolor="#ffcccc"];
43 * NOASDU [label="{noASDU (0x80)|Number of ASDUs: 1}", fillcolor="#ccffcc"];
44 * SEQASDU [label="{seqASDU (0xA2)|Sequence of ASDUs}", fillcolor="#ccccff"];
45 * ASDU [label="{ASDU (0x30)|One measurement sample}", fillcolor="#ffffcc"];
46 * SVID [label="{svID (0x80)|\"MUnn01\"}", fillcolor="#ffccff"];
47 * SMPCNT [label="{smpCnt (0x82)|Sample counter}", fillcolor="#ccffff"];
48 * CONFREV [label="{confRev (0x83)|Config revision}", fillcolor="#ffccff"];
49 * SMPSYNCH [label="{smpSynch (0x85)|Sync status}", fillcolor="#ccffff"];
50 * SEQDATA [label="{seqData (0x87)|8 channels × 8 bytes}", fillcolor="#ffffcc"];
51 *
52 * SAVPDU -> NOASDU;
53 * SAVPDU -> SEQASDU;
54 * SEQASDU -> ASDU;
55 * ASDU -> SVID;
56 * ASDU -> SMPCNT;
57 * ASDU -> CONFREV;
58 * ASDU -> SMPSYNCH;
59 * ASDU -> SEQDATA;
60 * }
61 * @enddot
62 *
63 * @section ber_length_encoding Length Encoding
64 *
65 * BER uses variable-length encoding for efficiency:
66 *
67 * | Value Range | Bytes | Format |
68 * |-------------|-------|--------|
69 * | 0-127 | 1 | Direct value |
70 * | 128-255 | 2 | 0x81 + 1 byte |
71 * | 256-65535 | 3 | 0x82 + 2 bytes |
72 * | 65536+ | 4-5 | 0x83/0x84 + bytes |
73 */

167/**
168 * @brief Calculate bytes needed to encode a BER length field
169 *
170 * BER uses variable-length encoding for the length field:
171 * - Length < 128: 1 byte (short form)
172 * - Length < 256: 2 bytes (0x81 + 1 byte)
173 * - Length < 65536: 3 bytes (0x82 + 2 bytes)
174 *
175 * @param length The length value to encode
176 * @return Number of bytes needed (1-5)
177 *
178 * @code
179 * size_t len_bytes = ber_length_of_length(200); // Returns 2
180 * @endcode
181 */
182size_t ber_length_of_length(size_t length)
183{
184 if (length < 128U) {
185 return 1U;
186 } else if (length <= 0xFFU) {
187 return 2U;
188 } else if (length <= 0xFFFFU) {
189 return 3U;
190 } else if (length <= 0xFFFFFFU) {
191 return 4U;
192 } else {
193 return 5U;
194 }
195}

214/**
215 * @brief Encode a BER length field
216 *
217 * Encodes the length using BER definite-length encoding:
218 * - Short form (1 byte) for length < 128
219 * - Long form (2-5 bytes) for larger lengths
220 *
221 * @param[out] buf Output buffer
222 * @param[in] buflen Size of output buffer
223 * @param[in] length Length value to encode
224 * @return Number of bytes written, or 0 if buffer too small
225 */
226size_t ber_encode_length(uint8_t *buf, size_t buflen, size_t length)
227{
228 if (length < 128U) {
229 if (buflen < 1U) return 0U;
230 buf[0] = (uint8_t)length;
231 return 1U;
232 } else if (length <= 0xFFU) {
233 if (buflen < 2U) return 0U;
234 buf[0] = 0x81U;
235 buf[1] = (uint8_t)length;
236 return 2U;
237 } else if (length <= 0xFFFFU) {
238 if (buflen < 3U) return 0U;
239 buf[0] = 0x82U;
240 buf[1] = (uint8_t)(length >> 8);
241 buf[2] = (uint8_t)(length & 0xFFU);
242 return 3U;
243 } else if (length <= 0xFFFFFFU) {
244 if (buflen < 4U) return 0U;
245 buf[0] = 0x83U;
246 buf[1] = (uint8_t)(length >> 16);
247 buf[2] = (uint8_t)((length >> 8) & 0xFFU);
248 buf[3] = (uint8_t)(length & 0xFFU);
249 return 4U;
250 } else {
251 if (buflen < 5U) return 0U;
252 buf[0] = 0x84U;
253 buf[1] = (uint8_t)(length >> 24);
254 buf[2] = (uint8_t)((length >> 16) & 0xFFU);
255 buf[3] = (uint8_t)((length >> 8) & 0xFFU);
256 buf[4] = (uint8_t)(length & 0xFFU);
257 return 5U;
258 }
259}

261/**
262 * @brief Encode a complete TLV (Tag-Length-Value) structure
263 *
264 * This is the core encoding function that combines tag, length, and value.
265 *
266 * @param[out] buf Output buffer
267 * @param[in] buflen Size of output buffer
268 * @param[in] tag Tag byte
269 * @param[in] value Value data (can be NULL if value_len is 0)
270 * @param[in] value_len Length of value data
271 * @return Total bytes written, or 0 if buffer too small
272 */
273size_t ber_encode_tlv(uint8_t *buf, size_t buflen, uint8_t tag,
274 const uint8_t *value, size_t value_len)
275{
276 size_t tag_len = 1U;
277 size_t len_len = ber_length_of_length(value_len);
278 size_t total = tag_len + len_len + value_len;
279
280 if (buflen < total) return 0U;
281
282 size_t offset = 0U;
283 offset += ber_encode_tag(buf + offset, buflen - offset, tag);
284 offset += ber_encode_length(buf + offset, buflen - offset, value_len);
285
286 if (value_len > 0U && value != NULL) {
287 memcpy(buf + offset, value, value_len);
288 offset += value_len;
289 }
290
291 return offset;
292}

294/**
295 * @brief Encode an unsigned integer
296 *
297 * Encodes using minimum bytes needed (BER DER rules).
298 *
299 * @param[out] buf Output buffer
300 * @param[in] buflen Size of output buffer
301 * @param[in] tag Tag byte
302 * @param[in] value Unsigned value to encode
303 * @return Bytes written, or 0 if buffer too small
304 */
305size_t ber_encode_unsigned(uint8_t *buf, size_t buflen, uint8_t tag, uint64_t value)
306{
307 uint8_t temp[9];
308 size_t len = 0U;
309
310 if (value == 0U) {
311 temp[0] = 0x00U;
312 len = 1U;
313 } else {
314 uint64_t v = value;
315 size_t nbytes = 0U;
316 while (v > 0U) {
317 nbytes++;
318 v >>= 8;
319 }
320
321 uint8_t msb = (uint8_t)(value >> ((nbytes - 1U) * 8U));
322 if (msb & 0x80U) {
323 temp[len++] = 0x00U;
324 }
325
326 for (size_t i = nbytes; i > 0U; i--) {
327 temp[len++] = (uint8_t)(value >> ((i - 1U) * 8U));
328 }
329 }
330
331 return ber_encode_tlv(buf, buflen, tag, temp, len);
332}

334/**
335 * @brief Encode a signed integer
336 *
337 * @param[out] buf Output buffer
338 * @param[in] buflen Size of output buffer
339 * @param[in] tag Tag byte
340 * @param[in] value Signed value to encode
341 * @return Bytes written, or 0 if buffer too small
342 */
343size_t ber_encode_signed(uint8_t *buf, size_t buflen, uint8_t tag, int64_t value)
344{
345 uint8_t temp[8];
346 size_t len = 0U;
347
348 if (value >= 0) {
349 return ber_encode_unsigned(buf, buflen, tag, (uint64_t)value);
350 } else {
351 int64_t v = value;
352 size_t nbytes = 1U;
353
354 while (nbytes < 8U) {
355 int64_t min_val = -((int64_t)1 << (nbytes * 8U - 1U));
356 if (v >= min_val) break;
357 nbytes++;
358 }
359
360 for (size_t i = nbytes; i > 0U; i--) {
361 temp[len++] = (uint8_t)(v >> ((i - 1U) * 8U));
362 }
363 }
364
365 return ber_encode_tlv(buf, buflen, tag, temp, len);
366}

368/**
369 * @brief Encode a 32-bit signed integer as fixed 4 bytes
370 *
371 * Used for SV sample values which are always exactly 4 bytes
372 * (no TLV wrapper, just raw big-endian bytes).
373 *
374 * @param[out] buf Output buffer (must be at least 4 bytes)
375 * @param[in] buflen Size of output buffer
376 * @param[in] value Value to encode
377 * @return 4 on success, 0 if buffer too small
378 */
379size_t ber_encode_int32_fixed(uint8_t *buf, size_t buflen, int32_t value)
380{
381 if (buflen < 4U) return 0U;
382
383 buf[0] = (uint8_t)((value >> 24) & 0xFFU);
384 buf[1] = (uint8_t)((value >> 16) & 0xFFU);
385 buf[2] = (uint8_t)((value >> 8) & 0xFFU);
386 buf[3] = (uint8_t)(value & 0xFFU);
387
388 return 4U;
389}
