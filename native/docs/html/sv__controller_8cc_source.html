1/**
2 * @file sv_controller.cc
3 * @brief Implementation of SharedBuffer + SvController + Multi-Publisher FFI
4 *
5 * This is the core of the multi-publisher architecture:
6 *
7 * 1. SharedBuffer::buildFromPublishers()
8 * - Takes all publishers' internal buffers
9 * - Merges into one sorted, interleaved schedule
10 * - Staggered timestamps so packets interleave evenly
11 *
12 * 2. SvController
13 * - Creates/manages SvPublisherInstance objects
14 * - On startAll(): prebuild → merge → spawn writer thread
15 * - Writer thread iterates the SharedBuffer, sends via npcap
16 *
17 * 3. FFI functions (extern "C")
18 * - sv_mp_* functions for Rust/Tauri to call
19 * - Complete lifecycle: add → configure → set equations → start → stop
20 *
21 * Architecture flow:
22 * UI → Tauri IPC → Rust FFI → sv_mp_* → SvController → SharedBuffer → npcap
23 */

47void SharedBuffer::buildFromPublishers(
48 const std::vector<std::unique_ptr<SvPublisherInstance>>& publishers)
49{
50 m_schedule.clear();
51 m_cycleDuration_us = 0;
52
53 /* Count READY publishers and total frames */
54 int readyCount = 0;
55 size_t totalFrames = 0;
56 for (const auto& pub : publishers) {
57 if (pub->getState() == SvPublisherInstance::READY) {
58 totalFrames += (size_t)pub->getFrameCount();
59 readyCount++;
60 }
61 }
62
63 if (readyCount == 0 || totalFrames == 0) {
64 printf("[shared-buffer] No ready publishers\n");
65 return;
66 }
67
68 m_schedule.reserve(totalFrames);
69
70 /*
71 * Stagger publishers evenly across time.
72 *
73 * If Publisher 1 sends every 250µs and Publisher 2 sends every 250µs,
74 * we stagger Publisher 2 by 125µs so the combined timeline is:
75 *
76 * t= 0µs: P1 frame 0
77 * t=125µs: P2 frame 0
78 * t=250µs: P1 frame 1
79 * t=375µs: P2 frame 1
80 * ...
81 *
82 * This distributes load evenly and avoids packet bursts.
83 */
84
85 /* Find the shortest interval (for stagger calculation) */
86 uint64_t minInterval_us = UINT64_MAX;
87 for (const auto& pub : publishers) {
88 if (pub->getState() != SvPublisherInstance::READY) continue;
89 uint64_t pps = pub->getPacketsPerSec();
90 if (pps == 0) continue;
91 uint64_t interval = 1000000ULL / pps;
92 if (interval < minInterval_us) minInterval_us = interval;
93 }
94 if (minInterval_us == UINT64_MAX) minInterval_us = 250;
95
96 uint64_t staggerStep = minInterval_us / (uint64_t)readyCount;
97 int pubIndex = 0;
98
99 for (const auto& pub : publishers) {
100 if (pub->getState() != SvPublisherInstance::READY) continue;
101
102 uint64_t pps = pub->getPacketsPerSec();
103 uint64_t interval_us = (pps > 0) ? (1000000ULL / pps) : 250;
104 uint64_t offset_us = (uint64_t)pubIndex * staggerStep;
105
106 for (int i = 0; i < pub->getFrameCount(); i++) {
107 ScheduleEntry entry;
108 entry.timestamp_us = offset_us + (uint64_t)i * interval_us;
109 entry.framePtr = pub->getFrame(i);
110 entry.frameLen = (uint16_t)pub->getFrameLen(i);
111 entry.publisherId = pub->getId();
112 m_schedule.push_back(entry);
113
114 /* Track max timestamp for cycle duration */
115 if (entry.timestamp_us + interval_us > m_cycleDuration_us)
116 m_cycleDuration_us = entry.timestamp_us + interval_us;
117 }
118
119 pubIndex++;
120 }
121
122 /* Sort by timestamp — this creates the interleaved playback order */
123 std::sort(m_schedule.begin(), m_schedule.end(),
124 [](const ScheduleEntry& a, const ScheduleEntry& b) {
125 if (a.timestamp_us != b.timestamp_us)
126 return a.timestamp_us < b.timestamp_us;
127 return a.publisherId < b.publisherId; /* stable tie-break */
128 });
129
130 printf("[shared-buffer] Built schedule: %zu entries from %d publishers, "
131 "cycle: %llu us\n",
132 m_schedule.size(), readyCount,
133 (unsigned long long)m_cycleDuration_us);
134}

341int SvController::startAll()
342{
343 if (m_running.load()) {
344 snprintf(m_errorBuf, sizeof(m_errorBuf), "Already running");
345 return -1;
346 }
347 if (npcap_publisher_is_running()) {
348 snprintf(m_errorBuf, sizeof(m_errorBuf),
349 "Single-publisher is already running. Stop it before using multi-publisher.");
350 return -1;
351 }
352 if (!npcap_is_open()) {
353 snprintf(m_errorBuf, sizeof(m_errorBuf), "No network interface open");
354 return -1;
355 }
356
357 std::lock_guard<std::mutex> lock(m_mutex);
358
359 if (m_publishers.empty()) {
360 snprintf(m_errorBuf, sizeof(m_errorBuf), "No publishers added");
361 return -1;
362 }
363
364 printf("[controller] ═══════════════════════════════════════════\n");
365 printf("[controller] Starting %zu publisher(s)...\n", m_publishers.size());
366
367 /*
368 * STEP 1: Pre-build frames for each publisher (SEQUENTIAL)
369 * Each publisher sets the global encoder config and builds its frames.
370 * Sequential execution avoids encoder state conflicts.
371 */
372 for (auto& pub : m_publishers) {
373 if (pub->getState() < SvPublisherInstance::CONFIGURED) {
374 printf("[controller] WARNING: publisher %u not configured, skipping\n",
375 pub->getId());
376 continue;
377 }
378 int ret = pub->prebuildFrames();
379 if (ret != 0) {
380 snprintf(m_errorBuf, sizeof(m_errorBuf),
381 "Publisher %u: prebuild failed: %s",
382 pub->getId(), pub->getLastError());
383 return -1;
384 }
385 }
386
387 /*
388 * STEP 2: Build SharedBuffer (merged interleaved schedule)
389 * Merges all publishers' internal buffers into one sorted timeline.
390 */
391 m_sharedBuffer.buildFromPublishers(m_publishers);
392
393 if (m_sharedBuffer.empty()) {
394 snprintf(m_errorBuf, sizeof(m_errorBuf),
395 "Shared buffer is empty — no frames built");
396 return -1;
397 }
398
399 /*
400 * STEP 3: Start writer thread
401 */
402 m_startTimeMs.store(npcap_stats_get_time_ms());
403 m_durationComplete.store(false);
404 m_repeatCycle.store(0);
405 m_running.store(true);
406
407 m_writerThread = std::thread(&SvController::writerLoop, this);
408
409 printf("[controller] Writer thread started\n");
410 printf("[controller] ═══════════════════════════════════════════\n");
411 return 0;
412}

414int SvController::stopAll()
415{
416 if (!m_running.load()) return 0;
417
418 printf("[controller] Stopping...\n");
419 m_running.store(false);
420
421 if (m_writerThread.joinable())
422 m_writerThread.join();
423
424 m_sharedBuffer.clear();
425
426 /* Defense-in-depth: clear all publishers so stale state never persists.
427 * The frontend always does removeAll + add + configure before each start,
428 * so keeping old publishers serves no purpose and risks accumulation bugs. */
429 {
430 std::lock_guard<std::mutex> lock(m_mutex);
431 m_publishers.clear();
432 m_nextId = 1;
433 }
434
435 printf("[controller] Stopped (publishers cleared)\n");
436 return 0;
437}

440{
441 printf("[controller] ═══════════════════════════════════════════\n");
442 printf("[controller] FULL RESET — clearing all state\n");
443
444 /* 1. Stop transmission if running */

459 }
460
461 /* 4. Reset global settings to defaults */
462 m_sendMode = 0;
463 m_durationSeconds = 0;
464 m_repeatEnabled = false;
465 m_repeatInfinite = false;
466 m_repeatCount = 0;
467 m_repeatCycle.store(0);

469 m_startTimeMs.store(0);
470 m_usbPadSize = 0;
471 m_usbMinGapUs = 0;
472 m_errorBuf[0] = '\0';
473
474 /* 5. Reset stats */
475 npcap_stats_reset();
476
477 /* 6. Stop single-publisher if it was running */
478 extern int npcap_publisher_stop(void);
479 npcap_publisher_stop();
480
481 printf("[controller] Reset complete — all memory freed, all state cleared\n");
482 printf("[controller] ═══════════════════════════════════════════\n");
483 return 0;
484}

486/*============================================================================
487 * Writer Thread — Entry Point
488 * Dispatches to batch or immediate mode based on sendMode setting.
489 *============================================================================*/
490
491void SvController::writerLoop()
492{
493 bool useBatch = false;
494
495 switch (m_sendMode) {
496 case 1: /* Force batch */
497 useBatch = npcap_sendqueue_available();
498 if (!useBatch)
499 printf("[controller] WARNING: batch mode requested but sendqueue "
500 "not available, falling back to immediate\n");
501 break;
502 case 2: /* Force immediate */
503 useBatch = false;
504 break;
505 case 3: /* USB-Optimized: force immediate + spin pacing */
506 useBatch = false;
507 break;
508 case 0: /* Auto */
509 default:
510 useBatch = npcap_sendqueue_available();
511 break;
512 }
513
514 if (useBatch)
515 writerLoopBatch();
516 else
517 writerLoopImmediate();
518}
519
520/*============================================================================
521 * Writer Loop — Batch Mode (sendqueue)
522 *
523 * Two-tier strategy (same as original single-publisher):
524 *
525 * TIER 1 — PRECISE (aggregate ≤4800 pps)
526 * 1 pkt/sendqueue, sync=1.
527 *
528 * TIER 2 — HIGH-SPEED (aggregate >4800 pps)
529 * Multi-pkt/sendqueue, sync=0 + QPC spin-wait pacing.
530 *============================================================================*/
531
532void SvController::writerLoopBatch()
533{
534 printf("[controller] Writer: BATCH mode (sendqueue)\n");
535
536 elevateThreadPriority();
537 npcap_stats_reset();
538 npcap_stats_session_start();
539
540 const size_t schedSize = m_sharedBuffer.size();
541 const uint64_t cycle_us = m_sharedBuffer.getCycleDuration();
542 const size_t frameLen = m_sharedBuffer[0].frameLen; /* approx, for sizing */
543
544 /* Aggregate rate across all publishers */
545 uint64_t aggregate_pps = (cycle_us > 0)
546 ? (schedSize * 1000000ULL / cycle_us) : schedSize;
547
548 /* Wire time per frame at 1 Gbps */
549 double wire_time_us = (double)(frameLen + 20) * 8.0 / 1000.0;
550
551 printf("[controller] Schedule: %zu entries, cycle: %llu us, "
552 "aggregate: %llu pps\n",
553 schedSize, (unsigned long long)cycle_us,
554 (unsigned long long)aggregate_pps);
555
556 uint64_t totalPackets = 0;
557
558 /*--------------------------------------------------------------
559 * TIER 1 — PRECISE (≤4800 pps aggregate)
560 *--------------------------------------------------------------*/
561 if (aggregate_pps <= 4800) {
562 printf("[controller] Tier: PRECISE (1 pkt/queue, sync=1)\n");
563
564 while (m_running.load() && !checkDurationElapsed()) {
565 uint64_t ts_accum = 0;
566
567 for (size_t i = 0; i < schedSize; i++) {
568 if (!m_running.load() || checkDurationElapsed()) break;
569
570 const ScheduleEntry& e = m_sharedBuffer[i];
571 unsigned int qsize = (unsigned int)(e.frameLen + 24 + 4096);
572 void* q = npcap_queue_create(qsize);
573 if (!q) continue;
574
575 npcap_queue_add(q, e.framePtr, e.frameLen, ts_accum);
576
577 /* Interval to NEXT entry */
578 if (i + 1 < schedSize) {
579 ts_accum += m_sharedBuffer[i + 1].timestamp_us
580 - e.timestamp_us;
581 } else {
582 /* Wrap: end of cycle → start of next cycle */
583 ts_accum += cycle_us - e.timestamp_us;
584 }
585
586 unsigned int sent = npcap_queue_transmit(q, 1);
587 if (sent > 0) {
588 totalPackets++;
589 npcap_stats_record_packet(e.frameLen);
590 }
591 npcap_queue_destroy(q);
592 }
593 }
594 }
595 /*--------------------------------------------------------------
596 * TIER 2 — HIGH-SPEED (>4800 pps aggregate)
597 *--------------------------------------------------------------*/
598 else {
599 /* Adaptive batch: target ~500µs of wire time per burst */
600 uint64_t batchSize = (uint64_t)(500.0 / wire_time_us);
601 if (batchSize < 50) batchSize = 50;
602 if (batchSize > 5000) batchSize = 5000;
603 if (batchSize > schedSize) batchSize = schedSize;
604
605 double batchInterval_us = (double)batchSize * 1000000.0
606 / (double)aggregate_pps;
607 auto batchDuration = std::chrono::nanoseconds(
608 (int64_t)(batchInterval_us * 1000.0));
609
610 printf("[controller] Tier: HIGH-SPEED (batch=%llu, sync=0, "
611 "QPC pacing)\n", (unsigned long long)batchSize);
612
613 unsigned int qmemsize =
614 (unsigned int)(batchSize * (frameLen + 24) + 4096);
615 auto nextBatch = std::chrono::high_resolution_clock::now();
616 uint64_t batchCount = 0;
617 uint64_t schedIdx = 0;
618
619 while (m_running.load() && !checkDurationElapsed()) {
620 void* q = npcap_queue_create(qmemsize);
621 if (!q) {
622 std::this_thread::sleep_for(std::chrono::milliseconds(1));
623 continue;
624 }
625
626 for (uint64_t b = 0; b < batchSize; b++) {
627 const ScheduleEntry& e =
628 m_sharedBuffer[schedIdx % schedSize];
629 npcap_queue_add(q, e.framePtr, e.frameLen, 0);
630 schedIdx++;
631 }
632
633 unsigned int sent = npcap_queue_transmit(q, 0);
634 if (sent > 0) {
635 uint32_t pkts = sent / (uint32_t)frameLen;
636 totalPackets += pkts;
637 for (uint32_t p = 0; p < pkts; p++)
638 npcap_stats_record_packet(frameLen);
639 }
640 npcap_queue_destroy(q);
641
642 /* QPC spin-wait for pacing */
643 nextBatch += batchDuration;
644 while (std::chrono::high_resolution_clock::now() < nextBatch)
645 spinPause();
646
647 /* Re-anchor every 200 batches to prevent drift */
648 batchCount++;
649 if ((batchCount % 200) == 0)
650 nextBatch = std::chrono::high_resolution_clock::now();
651 }
652 }
653
654 restoreThreadPriority();
655
656 /* Handle repeat */
657 if (checkDurationElapsed() && m_repeatEnabled) {
658 m_repeatCycle.fetch_add(1);
659 uint32_t cycle = m_repeatCycle.load();
660 if (m_repeatInfinite || cycle < m_repeatCount) {
661 printf("[controller] Repeat cycle %u\n", cycle + 1);
662 m_startTimeMs.store(npcap_stats_get_time_ms());
663 m_durationComplete.store(false);
664 writerLoopBatch();
665 return;
666 }
667 }
668
669 npcap_stats_session_end();
670
671 if (checkDurationElapsed()) {
672 m_durationComplete.store(true);
673 m_running.store(false);
674 }
675
676 printf("[controller] Batch writer complete: %llu packets\n",
677 (unsigned long long)totalPackets);
678}

680/*============================================================================
681 * Writer Loop — Immediate Mode (pcap_sendpacket per packet)
682 *
683 * Pacing modes (same as original single-publisher):
684 * ≤4800 pps: SLEEP + SPIN hybrid
685 * ≤50000 pps: pure SPIN (_mm_pause)
686 * >50000 pps: no pacing (max throughput)
687 *============================================================================*/
688
689void SvController::writerLoopImmediate()
690{
691 printf("[controller] Writer: IMMEDIATE mode (pcap_sendpacket)\n");
692
693 elevateThreadPriority();
694 npcap_stats_reset();
695 npcap_stats_session_start();
696
697 const size_t schedSize = m_sharedBuffer.size();
698 const uint64_t cycle_us = m_sharedBuffer.getCycleDuration();
699
700 uint64_t aggregate_pps = (cycle_us > 0)
701 ? (schedSize * 1000000ULL / cycle_us) : schedSize;
702
703 /* Choose pacing */
704 enum Pacing { SLEEP, SPIN, NONE };
705 Pacing pacing;
706 bool usbMode = (m_sendMode == 3);
707
708 if (usbMode) pacing = SPIN; /* USB: always spin */
709 else if (aggregate_pps <= 4800) pacing = SLEEP;
710 else if (aggregate_pps <= 50000) pacing = SPIN;
711 else pacing = NONE;
712
713 const char* pacingNames[] = {"SLEEP+SPIN", "SPIN", "NONE"};
714 printf("[controller] Pacing: %s, aggregate: %llu pps%s\n",
715 pacingNames[pacing], (unsigned long long)aggregate_pps,
716 usbMode ? " [USB-OPTIMIZED]" : "");
717
718 int usbPadSize = m_usbPadSize;
719 int usbGapUs = (m_usbMinGapUs > 0) ? m_usbMinGapUs : 130;
720 if (usbMode) {
721 printf("[controller] USB-OPTIMIZED: gap=%d us", usbGapUs);
722 if (usbPadSize > 0)
723 printf(", padding=%d bytes", usbPadSize);
724 printf("\n");
725 }
726
727 uint64_t totalPackets = 0;
728 uint64_t totalFailures = 0;
729 uint64_t schedIdx = 0;
730
731 /* Calculate average interval between consecutive schedule entries */
732 double avg_interval_us = (aggregate_pps > 0)
733 ? (1000000.0 / (double)aggregate_pps) : 250.0;
734 auto intervalDur = std::chrono::nanoseconds(
735 (int64_t)(avg_interval_us * 1000.0));
736
737 /* Epoch-based absolute scheduling (industry standard).
738 *
739 * target[N] = epoch + N * interval
740 *
741 * This ensures the average packet rate is EXACTLY correct regardless
742 * of how long each pcap_sendpacket() call takes.
743 *
744 * If pcap_sendpacket takes longer than the interval, we fall behind
745 * schedule. To handle this:
746 * - Small lag: send next packet immediately (natural catch-up)
747 * - Large lag (>3 intervals behind): re-anchor epoch to prevent
748 * unbounded bursts of back-to-back packets
749 *
750 * This is the same approach used by tcpreplay, Linux pktgen, and
751 * professional traffic generators (DPDK TRex, MoonGen, etc.).
752 */
753 auto epoch = std::chrono::high_resolution_clock::now();
754 uint64_t pktNum = 0;
755
756 while (m_running.load() && !checkDurationElapsed()) {
757 const ScheduleEntry& e = m_sharedBuffer[schedIdx % schedSize];
758
759 int result;
760 if (usbMode && usbPadSize > 0 && usbPadSize > (int)e.frameLen) {
761 result = npcap_send_packet_padded(e.framePtr, e.frameLen, (size_t)usbPadSize);
762 } else {
763 result = npcap_send_packet(e.framePtr, e.frameLen);
764 }
765 if (result == 0) {
766 totalPackets++;
767 npcap_stats_record_packet(e.frameLen);
768 } else {
769 totalFailures++;
770 npcap_stats_record_failure();
771 }
772
773 schedIdx++;
774 pktNum++;
775
776 if (pacing == NONE)
777 continue;
778
779 /* Absolute target for the NEXT packet */
780 auto target = epoch + pktNum * intervalDur;
781 auto now = std::chrono::high_resolution_clock::now();
782
783 if (target <= now) {
784 /* Behind schedule — check how far */
785 auto behind = std::chrono::duration_cast<
786 std::chrono::nanoseconds>(now - target);
787 if (behind > intervalDur) {
788 /* Behind by >1 interval — re-anchor to prevent burst.
789 * Allows at most 1 catch-up packet (small lag < 1 interval)
790 * but prevents back-to-back bursts of 3-4 packets that
791 * appear as 0.000000 deltas in Wireshark. */
792 epoch = now;
793 pktNum = 0;
794 }
795
796 /* USB mode: even when behind, enforce minimum gap between
797 * sends so the USB host controller doesn't batch packets
798 * into the same microframe (125µs). Configurable gap. */
799 if (usbMode) {
800 auto minTarget = now + std::chrono::microseconds(usbGapUs);
801 while (std::chrono::high_resolution_clock::now() < minTarget)
802 spinPause();
803 }
804 continue;
805 }
806
807 /* Ahead of schedule — pace to exact target */
808 switch (pacing) {
809 case SLEEP: {
810 auto remaining = std::chrono::duration_cast<
811 std::chrono::microseconds>(target - now);
812 if (remaining.count() > 80)
813 std::this_thread::sleep_for(
814 remaining - std::chrono::microseconds(80));
815 while (std::chrono::high_resolution_clock::now() < target)
816 spinPause();
817 break;
818 }
819 case SPIN:
820 while (std::chrono::high_resolution_clock::now() < target)
821 spinPause();
822 break;
823 default:
824 break;
825 }
826 }
827
828 restoreThreadPriority();
829
830 /* Handle repeat */
831 if (checkDurationElapsed() && m_repeatEnabled) {
832 m_repeatCycle.fetch_add(1);
833 uint32_t cycle = m_repeatCycle.load();
834 if (m_repeatInfinite || cycle < m_repeatCount) {
835 printf("[controller] Repeat cycle %u\n", cycle + 1);
836 m_startTimeMs.store(npcap_stats_get_time_ms());
837 m_durationComplete.store(false);
838 writerLoopImmediate();
839 return;
840 }
841 }
842
843 npcap_stats_session_end();
844
845 if (checkDurationElapsed()) {
846 m_durationComplete.store(true);
847 m_running.store(false);
848 }
849
850 printf("[controller] Immediate writer complete: %llu sent, %llu failed\n",
851 (unsigned long long)totalPackets,
852 (unsigned long long)totalFailures);
853}
