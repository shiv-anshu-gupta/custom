1/**
2 * @file sv_controller.h
3 * @brief Main Controller + SharedBuffer for Multi-Publisher SV System
4 *
5 * Architecture:
6 *
7 * SvController (Main Class)
8 * │
9 * ├── creates SvPublisherInstance 1 ──→ Internal Buffer 1
10 * ├── creates SvPublisherInstance 2 ──→ Internal Buffer 2
11 * └── creates SvPublisherInstance N ──→ Internal Buffer N
12 * │
13 * ┌───────────────────────┘
14 * ▼
15 * SharedBuffer (merged schedule)
16 * Sorted by timestamp, interleaved
17 * │
18 * ▼
19 * Writer Thread (single)
20 * TIME_CRITICAL priority
21 * pcap_sendpacket / sendqueue
22 * │
23 * ▼
24 * Network / NIC
25 *
26 * ## SharedBuffer
27 * The SharedBuffer merges all publishers' internal buffers into a single
28 * time-ordered interleaved schedule. Each entry points directly into the
29 * publisher's pre-built frame cache (zero copy).
30 *
31 * This serves the same architectural role as SharedRingBuffer (teammate's
32 * implementation), but optimized for the pre-built SV use case:
33 * - No Boost dependency
34 * - No locking at runtime (schedule is immutable once built)
35 * - Zero-copy (pointers into publishers' frame caches)
36 * - Timestamp-ordered (sorted during build)
37 *
38 * @note When SharedRingBuffer with Boost.Interprocess is available, the
39 * SharedBuffer can be replaced without changing the rest of the
40 * architecture.
41 */

68/*============================================================================
69 * ScheduleEntry — one slot in the shared buffer
70 *============================================================================*/
71
72struct ScheduleEntry {
73 uint64_t timestamp_us; /* When to transmit (relative to cycle start) */
74 uint8_t* framePtr; /* Direct pointer into publisher's frame cache */
75 uint16_t frameLen; /* Frame length in bytes */
76 uint32_t publisherId; /* Which publisher owns this frame */
77};
