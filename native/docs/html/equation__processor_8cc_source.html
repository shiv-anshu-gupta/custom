24static void trim(char* str) {
25 if (!str || !*str) return;
26
27 char* start = str;
28 while (*start && isspace((unsigned char)*start)) start++;
29
30 char* end = start + strlen(start) - 1;
31 while (end > start && isspace((unsigned char)*end)) *end-- = '\0';
32
33 if (start != str) memmove(str, start, strlen(start) + 1);
34}

36static const char* find_ci(const char* str, const char* pattern) {
37 if (!str || !pattern) return NULL;
38
39 size_t slen = strlen(str), plen = strlen(pattern);
40 if (plen > slen) return NULL;
41
42 for (size_t i = 0; i <= slen - plen; i++) {
43 int match = 1;
44 for (size_t j = 0; j < plen && match; j++) {
45 if (tolower((unsigned char)str[i+j]) != tolower((unsigned char)pattern[j]))
46 match = 0;
47 }
48 if (match) return str + i;
49 }
50 return NULL;
51}

57static double parse_amplitude(const char* eq) {
58 if (!eq) return 0.0;
59
60 const char* star = strchr(eq, '*');
61 if (!star) return atof(eq);
62
63 char buf[64];
64 size_t len = star - eq;
65 if (len >= sizeof(buf)) len = sizeof(buf) - 1;
66 strncpy(buf, eq, len);
67 buf[len] = '\0';
68 trim(buf);
69
70 return atof(buf);
71}

73static double parse_frequency(const char* eq, double defaultFreq) {
74 if (!eq) return defaultFreq;
75
76 /* Normalize string */
77 char norm[EQ_MAX_EQUATION_LEN];
78 size_t j = 0;
79 for (size_t i = 0; eq[i] && j < sizeof(norm) - 1; i++) {
80 if (!isspace((unsigned char)eq[i])) {
81 norm[j++] = tolower((unsigned char)eq[i]);
82 }
83 }
84 norm[j] = '\0';
85
86 /* Find "pi*" pattern */
87 const char* pi = strstr(norm, "pi*");
88 if (!pi) pi = strstr(norm, "pi ");
89 if (!pi) return defaultFreq;
90
91 const char* start = pi + 3;
92 while (*start == '*' || *start == ' ') start++;
93
94 const char* end = strstr(start, "*t");
95 if (!end) end = strchr(start, 't');
96 if (!end) end = strchr(start, ')');
97 if (!end) return defaultFreq;
98
99 char buf[32];
100 size_t len = end - start;
101 while (len > 0 && (start[len-1] == '*' || start[len-1] == ' ')) len--;
102 if (len >= sizeof(buf)) len = sizeof(buf) - 1;
103
104 strncpy(buf, start, len);
105 buf[len] = '\0';
106
107 double freq = atof(buf);
108 return (freq > 0 && freq < 1e9) ? freq : defaultFreq;
109}

111static double parse_phase(const char* eq) {
112 if (!eq) return 0.0;
113
114 char norm[EQ_MAX_EQUATION_LEN];
115 size_t j = 0;
116 for (size_t i = 0; eq[i] && j < sizeof(norm) - 1; i++) {
117 if (!isspace((unsigned char)eq[i])) {
118 norm[j++] = tolower((unsigned char)eq[i]);
119 }
120 }
121 norm[j] = '\0';
122
123 const char* t = strstr(norm, "*t");
124 if (!t) t = strchr(norm, 't');
125 if (!t) return 0.0;
126
127 const char* after = t + 1;
128 if (*after == '*') after++;
129
130 /* Common phase patterns */
131 if (strstr(after, "-2*pi/3") || strstr(after, "-2pi/3")) return -EQ_TWO_PI / 3.0;
132 if (strstr(after, "+2*pi/3") || strstr(after, "+2pi/3") || strstr(after, "2*pi/3")) return EQ_TWO_PI / 3.0;
133 if (strstr(after, "-4*pi/3") || strstr(after, "-4pi/3")) return -4.0 * EQ_PI / 3.0;
134 if (strstr(after, "+4*pi/3") || strstr(after, "+4pi/3")) return 4.0 * EQ_PI / 3.0;
135 if (strstr(after, "-pi/2")) return -EQ_PI / 2.0;
136 if (strstr(after, "+pi/2")) return EQ_PI / 2.0;
137
138 return 0.0;
139}

141static EqWaveformType parse_wavetype(const char* eq) {
142 if (!eq) return EQ_WAVE_SINE;
143 if (find_ci(eq, "cos(")) return EQ_WAVE_COSINE;

176int eq_parse_equation(const char* equation, EqChannelData* ch, double defaultFreq) {
177 if (!ch) return -1;
178
179 ch->amplitude = 0;
180 ch->frequency = defaultFreq;
181 ch->phaseOffset = 0;
182 ch->scaleFactor = 1000.0;
183 ch->isZero = 0;
184 ch->isValid = 0;
185 ch->waveType = EQ_WAVE_SINE;
186
187 if (!equation || !equation[0] || strcmp(equation, "0") == 0 || strcmp(equation, "0.0") == 0) {
188 ch->isZero = 1;
189 ch->isValid = 1;
190 return 0;
191 }
192
193 strncpy(ch->equation, equation, EQ_MAX_EQUATION_LEN - 1);
194 ch->equation[EQ_MAX_EQUATION_LEN - 1] = '\0';
195 trim(ch->equation);
196
197 ch->amplitude = parse_amplitude(ch->equation);
198 ch->frequency = parse_frequency(ch->equation, defaultFreq);
199 ch->phaseOffset = parse_phase(ch->equation);
200 ch->waveType = parse_wavetype(ch->equation);
201 ch->isValid = 1;
202
203 return 0;
204}

206int eq_load_equations(EqProcessor* proc, const char* equations) {
207 if (!proc || !equations) return -1;
208
209 eq_processor_reset(proc);
210
211 printf("[eq] Loading equations (freq=%.0f Hz, rate=%u)\n",
212 proc->defaultFrequency, proc->sampleRate);
213
214 char* str = strdup(equations);
215 if (!str) return -1;
216
217 char* saveptr;
218 char* token = strtok_r(str, "|", &saveptr);
219 int count = 0;
220
221 while (token && proc->channelCount < EQ_MAX_CHANNELS) {
222 char* colon = strchr(token, ':');
223 if (!colon) { token = strtok_r(NULL, "|", &saveptr); continue; }
224
225 *colon = '\0';
226 char* id = token;
227 char* eq = colon + 1;
228 trim(id);
229 trim(eq);
230
231 EqChannelData* ch = &proc->channels[proc->channelCount];
232 strncpy(ch->id, id, EQ_MAX_ID_LEN - 1);
233
234 if (eq_parse_equation(eq, ch, proc->defaultFrequency) == 0) {
235 printf("[eq] %s: amp=%.1f, freq=%.0f Hz, phase=%.1f°\n",
236 ch->id, ch->amplitude, ch->frequency, ch->phaseOffset * 180.0 / EQ_PI);
237 proc->channelCount++;
238 count++;
239 } else {
240 proc->parseErrors++;
241 }
242
243 token = strtok_r(NULL, "|", &saveptr);
244 }
245
246 free(str);
247 printf("[eq] Loaded %d channels\n", count);
248 return count;
249}

251int eq_set_channel_equation(EqProcessor* proc, const char* channelId, const char* equation) {
252 if (!proc || !channelId) return -1;
253
254 int idx = -1;
255 for (int i = 0; i < proc->channelCount; i++) {
256 if (strcmp(proc->channels[i].id, channelId) == 0) { idx = i; break; }

272int32_t eq_generate_sample(const EqChannelData* ch, double t) {
273 if (!ch || ch->isZero || !ch->isValid) return 0;
274
275 double omega = EQ_TWO_PI * ch->frequency;
276 double angle = omega * t + ch->phaseOffset;
277 double value = 0.0;
278
279 switch (ch->waveType) {
280 case EQ_WAVE_SINE: value = ch->amplitude * sin(angle); break;
281 case EQ_WAVE_COSINE: value = ch->amplitude * cos(angle); break;
282 case EQ_WAVE_SQUARE: value = ch->amplitude * (sin(angle) >= 0 ? 1.0 : -1.0); break;
283 case EQ_WAVE_TRIANGLE: {
284 double phase = fmod(angle, EQ_TWO_PI);
285 if (phase < 0) phase += EQ_TWO_PI;
286 value = ch->amplitude * (phase < EQ_PI ? (2.0 * phase / EQ_PI - 1.0) : (3.0 - 2.0 * phase / EQ_PI));
287 break;
288 }
289 }
290
291 return (int32_t)(value * ch->scaleFactor);
292}

371int eq_validate_equation(const char* eq) {
372 if (!eq || !eq[0] || strcmp(eq, "0") == 0) return 1;
373 if (find_ci(eq, "sin") || find_ci(eq, "cos") || find_ci(eq, "square") || find_ci(eq, "triangle")) return 1;
374
375 char* endptr;
376 strtod(eq, &endptr);
377 return (*endptr == '\0' || isspace((unsigned char)*endptr));
