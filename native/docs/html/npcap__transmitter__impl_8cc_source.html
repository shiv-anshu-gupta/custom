123
124 /* Try Npcap path first, then fallback to system path */
125 char path[MAX_PATH];
126 if (GetEnvironmentVariableA("SystemRoot", path, MAX_PATH)) {
127 strcat(path, "\\System32\\Npcap\\wpcap.dll");
128 g_dll = LoadLibraryA(path);
129 }
130 if (!g_dll) g_dll = LoadLibraryA("wpcap.dll");
131
132 if (!g_dll) {
133 snprintf(g_error, sizeof(g_error),

135 return 0;
136 }

180int npcap_list_interfaces(NpcapInterface* interfaces, int max_count) {
181 printf("[npcap] Listing network interfaces...\n");
182
183#ifdef _WIN32
184 if (!npcap_load_dll()) return -1;
185
186 pcap_if_t* alldevs = nullptr;
187 char errbuf[256] = {0};
188
189 if (g_findalldevs(&alldevs, errbuf) == -1) {
190 snprintf(g_error, sizeof(g_error), "pcap_findalldevs failed: %s", errbuf);
191 printf("[npcap] ERROR: %s\n", g_error);
192 return -1;
193 }
194
195 if (!alldevs) {
196 printf("[npcap] WARNING: No interfaces found by pcap_findalldevs\n");
197 return 0;
198 }
199
200 int count = 0;
201 for (pcap_if_t* d = alldevs; d && count < max_count; d = d->next) {
202 NpcapInterface* iface = &interfaces[count];
203 memset(iface, 0, sizeof(NpcapInterface));
204
205 if (d->name) {
206 strncpy(iface->name, d->name, sizeof(iface->name) - 1);
207 }
208 if (d->description) {
209 strncpy(iface->description, d->description, sizeof(iface->description) - 1);
210 }
211
212 /* Get MAC address from Windows adapter info */
213 PIP_ADAPTER_INFO adapterInfo = nullptr;
214 ULONG bufLen = 0;
215
216 if (GetAdaptersInfo(adapterInfo, &bufLen) == ERROR_BUFFER_OVERFLOW) {
217 adapterInfo = (PIP_ADAPTER_INFO)malloc(bufLen);
218 if (adapterInfo && GetAdaptersInfo(adapterInfo, &bufLen) == NO_ERROR) {
219 for (PIP_ADAPTER_INFO ai = adapterInfo; ai; ai = ai->Next) {
220 printf("[npcap] Checking: pcap='%s' vs adapter='%s'\n", d->name, ai->AdapterName);
221 if (d->name && strstr(d->name, ai->AdapterName)) {
222 memcpy(iface->mac, ai->Address, 6);
223 iface->has_mac = 1;
224 printf("[npcap] MAC found: %02X:%02X:%02X:%02X:%02X:%02X\n",
225 iface->mac[0], iface->mac[1], iface->mac[2],
226 iface->mac[3], iface->mac[4], iface->mac[5]);
227 break;
228 }
229 }
230 }
231 if (adapterInfo) free(adapterInfo);
232 }
233
234 printf("[npcap] Interface %d: %s (has_mac=%d)\n", count,
235 iface->description[0] ? iface->description : iface->name, iface->has_mac);
236 count++;
237 }
238
239 g_freealldevs(alldevs);
240 printf("[npcap] Total: %d interfaces found\n", count);
241 return count;
242#else
243 pcap_if_t* alldevs;
244 char errbuf[PCAP_ERRBUF_SIZE];
245
246 if (pcap_findalldevs(&alldevs, errbuf) == -1) {
247 snprintf(g_error, sizeof(g_error), "%s", errbuf);
248 return -1;
249 }
250
251 int count = 0;
252 for (pcap_if_t* d = alldevs; d && count < max_count; d = d->next) {
253 NpcapInterface* iface = &interfaces[count++];
254 memset(iface, 0, sizeof(NpcapInterface));
255 if (d->name) strncpy(iface->name, d->name, sizeof(iface->name) - 1);
256 if (d->description) strncpy(iface->description, d->description, sizeof(iface->description) - 1);
257 }
258 pcap_freealldevs(alldevs);
259 return count;
260#endif
261}

267int npcap_open(const char* device_name) {
268 printf("[npcap] Opening: %s\n", device_name);
269
270#ifdef _WIN32
271 if (!npcap_load_dll()) return -1;
272
273 if (g_handle) { g_close(g_handle); g_handle = nullptr; }
274
275 char errbuf[256];
276
277 /* Prefer pcap_create workflow to enable HOST_HIPREC timestamps.
278 * HOST_HIPREC uses QPC internally in Npcap kernel driver for
279 * microsecond-precision timestamps — no need for manual QPC. */
280 if (g_pcap_create && g_set_snaplen && g_set_promisc &&
281 g_set_timeout && g_set_tstamp_type && g_activate) {
282
283 g_handle = g_pcap_create(device_name, errbuf);
284 if (!g_handle) {
285 snprintf(g_error, sizeof(g_error), "pcap_create: %s", errbuf);
286 return -1;
287 }
288
289 g_set_snaplen(g_handle, 65536);
290 g_set_promisc(g_handle, 1);
291 g_set_timeout(g_handle, 1);
292
293 /* Set high-precision timestamp type (HOST_HIPREC = QPC-based) */
294 int tstamp_ret = g_set_tstamp_type(g_handle, PCAP_TSTAMP_HOST_HIPREC);
295 if (tstamp_ret == 0) {
296 printf("[npcap] HOST_HIPREC timestamp enabled\n");
297 } else {
298 printf("[npcap] HOST_HIPREC not supported (code=%d), using default\n", tstamp_ret);
299 }
300
301 /* Immediate mode: minimize driver-level send/receive buffering.
302 * Critical for USB Ethernet adapters to reduce packet batching. */
303 if (g_set_immediate_mode) {
304 g_set_immediate_mode(g_handle, 1);
305 printf("[npcap] Immediate mode enabled (reduced buffering)\n");
306 }
307
308 int activate_ret = g_activate(g_handle);
309 if (activate_ret < 0) {
310 snprintf(g_error, sizeof(g_error), "pcap_activate failed (code=%d)", activate_ret);
311 g_close(g_handle);
312 g_handle = nullptr;
313 return -1;
314 }
315 if (activate_ret > 0) {
316 printf("[npcap] pcap_activate warning (code=%d)\n", activate_ret);
317 }
318
319 printf("[npcap] Interface opened (pcap_create + HOST_HIPREC)\n");
320 } else {
321 /* Fallback to pcap_open_live if pcap_create not available */
322 g_handle = g_open_live(device_name, 65536, 1, 1, errbuf);
323 if (!g_handle) {
324 snprintf(g_error, sizeof(g_error), "pcap_open_live: %s", errbuf);
325 return -1;
326 }
327 printf("[npcap] Interface opened (pcap_open_live fallback)\n");
328 }
329#else
330 char errbuf[PCAP_ERRBUF_SIZE];
331 g_handle = pcap_open_live(device_name, 65536, 1, 1, errbuf);
332 if (!g_handle) {
333 snprintf(g_error, sizeof(g_error), "%s", errbuf);
334 return -1;
335 }
336 printf("[npcap] Interface opened\n");
337#endif
338
339 return 0;
340}

424/*============================================================================
425 * Frame Padding for USB-Ethernet
426 *
427 * Pad each SV frame with trailing zeros so fewer frames fit per USB bulk
428 * transfer, forcing the adapter driver to submit URBs sooner.
429 * IEC 61850-9-2 subscribers parse by PDU length fields, so trailing
430 * padding zeros are safely ignored by compliant devices.
431 *============================================================================*/
432
433int npcap_send_packet_padded(const uint8_t* data, size_t len, size_t pad_to) {
434#ifdef _WIN32
435 if (!g_handle || !g_sendpacket) return -1;
436
437 if (pad_to <= len || pad_to > 1522) {
438 return g_sendpacket(g_handle, data, (int)len);
439 }
440
441 uint8_t padded[1522];
442 memcpy(padded, data, len);
443 memset(padded + len, 0, pad_to - len);
444 return g_sendpacket(g_handle, padded, (int)pad_to);
445#else
446 if (!g_handle) return -1;
447 if (pad_to <= len || pad_to > 1522) {
448 return pcap_sendpacket(g_handle, data, (int)len);
449 }
450 uint8_t padded[1522];
451 memcpy(padded, data, len);
452 memset(padded + len, 0, pad_to - len);
453 return pcap_sendpacket(g_handle, padded, (int)pad_to);
454#endif
455}
