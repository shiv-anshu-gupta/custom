1/**
2 * @file sv_encoder_impl.cc
3 * @brief IEC 61850-9-2LE Sampled Values Packet Encoder
4 *
5 * This module encodes SV packets according to the IEC 61850-9-2LE standard.
6 *
7 * Packet Structure:
8 * - Ethernet Header (14 bytes, +4 with VLAN)
9 * - SV Header: AppID(2) + Length(2) + Reserved(4) = 8 bytes
10 * - savPdu: ASN.1 BER encoded payload
11 * - noASDU: Number of ASDUs
12 * - seqASDU: Sequence of ASDU blocks
13 * - ASDU: svID, smpCnt, confRev, smpSynch, seqData
14 *
15 * Key Formula (matching Node.js working version):
16 * APDU Length = savPduLen + 10 (single ASDU)
17 * APDU Length = savPduLen + 12 (multi-ASDU)
18 */
19

59/*============================================================================
60 * Single ASDU Encoder
61 *
62 * Encodes one ASDU per packet. Used for standard 4000/4800 Hz rates.
63 *============================================================================*/
64
65int sv_encoder_encode_packet(
66 uint32_t smpCnt,
67 const int32_t* samples,
68 uint8_t* buffer,
69 size_t* outSize
70) {
71 if (!buffer || !outSize || !samples) return -1;
72
73 std::lock_guard<std::mutex> lock(g_mutex);
74 size_t pos = 0;
75
76 /* Ethernet Header */
77 memcpy(buffer + pos, g_config.dstMAC, 6); pos += 6;
78 memcpy(buffer + pos, g_config.srcMAC, 6); pos += 6;
79
80 /* Optional VLAN Tag (802.1Q) */
81 if (g_config.vlanID > 0) {
82 buffer[pos++] = 0x81;
83 buffer[pos++] = 0x00;
84 uint16_t vlanTag = ((g_config.vlanPriority & 0x7) << 13) | (g_config.vlanID & 0xFFF);
85 buffer[pos++] = (vlanTag >> 8) & 0xFF;
86 buffer[pos++] = vlanTag & 0xFF;
87 }
88
89 /* EtherType: SV (0x88BA) */
90 buffer[pos++] = 0x88;
91 buffer[pos++] = 0xBA;
92
93 /* APPID */
94 buffer[pos++] = (g_config.appID >> 8) & 0xFF;
95 buffer[pos++] = g_config.appID & 0xFF;
96
97 /* Length placeholder */
98 size_t lengthPos = pos;
99 buffer[pos++] = 0;
100 buffer[pos++] = 0;
101
102 /* Reserved (4 bytes) */
103 buffer[pos++] = 0; buffer[pos++] = 0;
104 buffer[pos++] = 0; buffer[pos++] = 0;
105
106 /* savPdu (ASN.1 tag 0x60 = APPLICATION 0) */
107 buffer[pos++] = 0x60;
108 size_t savPduLenPos = pos;
109 buffer[pos++] = 0; /* Length placeholder */
110
111 /* noASDU (tag 0x80 = context [0]) */
112 buffer[pos++] = 0x80;
113 buffer[pos++] = 0x01;
114 buffer[pos++] = 0x01; /* 1 ASDU */
115
116 /* seqASDU (tag 0xA2 = context [2] constructed) */
117 buffer[pos++] = 0xA2;
118 size_t seqASDULenPos = pos;
119 buffer[pos++] = 0; /* Length placeholder */
120
121 /* ASDU (tag 0x30 = SEQUENCE) */
122 buffer[pos++] = 0x30;
123 size_t asduLenPos = pos;
124 buffer[pos++] = 0; /* Length placeholder */
125
126 /* svID (tag 0x80 = context [0]) */
127 size_t svIDLen = strlen(g_config.svID);
128 if (svIDLen == 0) {
129 /* Fallback to default if empty */
130 buffer[pos++] = 0x80;
131 buffer[pos++] = 0x04;
132 memcpy(buffer + pos, "MU01", 4); pos += 4;
133 svIDLen = 4;
134 } else {
135 buffer[pos++] = 0x80;
136 buffer[pos++] = (uint8_t)svIDLen;
137 memcpy(buffer + pos, g_config.svID, svIDLen); pos += svIDLen;
138 }
139
140 /* smpCnt (tag 0x82 = context [2]) */
141 buffer[pos++] = 0x82;
142 buffer[pos++] = 0x02;
143 buffer[pos++] = (smpCnt >> 8) & 0xFF;
144 buffer[pos++] = smpCnt & 0xFF;
145
146 /* confRev (tag 0x83 = context [3]) */
147 buffer[pos++] = 0x83;
148 buffer[pos++] = 0x04;
149 buffer[pos++] = (g_config.confRev >> 24) & 0xFF;
150 buffer[pos++] = (g_config.confRev >> 16) & 0xFF;
151 buffer[pos++] = (g_config.confRev >> 8) & 0xFF;
152 buffer[pos++] = g_config.confRev & 0xFF;
153
154 /* smpSynch (tag 0x85 = context [5]) */
155 buffer[pos++] = 0x85;
156 buffer[pos++] = 0x01;
157 buffer[pos++] = g_config.smpSynch;
158
159 /* seqData (tag 0x87 = context [7])
160 * Dynamic channels: channelCount × (4 bytes value + 4 bytes quality) */
161 uint8_t channelCount = g_config.channelCount;
162 uint8_t seqDataLen = channelCount * 8; // 8 bytes per channel
163 buffer[pos++] = 0x87;
164 buffer[pos++] = seqDataLen;
165
166 for (int i = 0; i < channelCount; i++) {
167 int32_t val = samples[i];
168 /* Value (big-endian) */
169 buffer[pos++] = (val >> 24) & 0xFF;
170 buffer[pos++] = (val >> 16) & 0xFF;
171 buffer[pos++] = (val >> 8) & 0xFF;
172 buffer[pos++] = val & 0xFF;
173 /* Quality (0x0000 = Good) */
174 buffer[pos++] = 0x00;
175 buffer[pos++] = 0x00;
176 buffer[pos++] = 0x00;
177 buffer[pos++] = 0x00;
178 }
179
180 /* Fill in lengths */
181 size_t asduLen = pos - asduLenPos - 1;
182 buffer[asduLenPos] = (uint8_t)asduLen;
183
184 size_t seqASDULen = pos - seqASDULenPos - 1;
185 buffer[seqASDULenPos] = (uint8_t)seqASDULen;
186
187 size_t savPduLen = pos - savPduLenPos - 1;
188 buffer[savPduLenPos] = (uint8_t)savPduLen;
189
190 /* APDU Length: savPduLen + 10 (matches Node.js working formula) */
191 size_t apduLen = savPduLen + 10;
192 buffer[lengthPos] = (apduLen >> 8) & 0xFF;
193 buffer[lengthPos + 1] = apduLen & 0xFF;
194
195 /* Debug output (first packet only) */
196 if (!g_debug_printed) {
197 printf("[encoder] First packet: svID=%zu bytes, channels=%d, ASDU=%zu, APDU=%zu\n",
198 svIDLen, channelCount, asduLen, apduLen);
199 g_debug_printed = true;
200 }
201
202 /* Pad to minimum Ethernet frame (60 bytes) */
203 while (pos < 60) buffer[pos++] = 0;
204
205 *outSize = pos;
206 return 0;
207}

222

224 size_t pos = 0;
225 uint8_t asduCount = g_config.asduCount;
226

228 memcpy(buffer + pos, g_config.dstMAC, 6); pos += 6;
229 memcpy(buffer + pos, g_config.srcMAC, 6); pos += 6;
230

232 if (g_config.vlanID > 0) {
233 buffer[pos++] = 0x81;
234 buffer[pos++] = 0x00;
235 uint16_t vlanTag = ((g_config.vlanPriority & 0x7) << 13) | (g_config.vlanID & 0xFFF);
236 buffer[pos++] = (vlanTag >> 8) & 0xFF;
237 buffer[pos++] = vlanTag & 0xFF;
238 }
239
240 /* EtherType: SV */
241 buffer[pos++] = 0x88;
242 buffer[pos++] = 0xBA;
243
244 /* APPID */
245 buffer[pos++] = (g_config.appID >> 8) & 0xFF;
246 buffer[pos++] = g_config.appID & 0xFF;
247
248 /* Length placeholder */
249 size_t lengthPos = pos;
250 buffer[pos++] = 0;
251 buffer[pos++] = 0;
252
253 /* Reserved */
254 buffer[pos++] = 0; buffer[pos++] = 0;
255 buffer[pos++] = 0; buffer[pos++] = 0;
256
257 /* savPdu with 2-byte length (for larger payloads) */
258 buffer[pos++] = 0x60;
259 size_t savPduLenPos = pos;
260 buffer[pos++] = 0x82; /* 2-byte length follows */
261 buffer[pos++] = 0;
262 buffer[pos++] = 0;
263
264 /* noASDU */
265 buffer[pos++] = 0x80;
266 buffer[pos++] = 0x01;
267 buffer[pos++] = asduCount;
268
269 /* seqASDU with 2-byte length */
270 buffer[pos++] = 0xA2;
271 size_t seqASDULenPos = pos;
272 buffer[pos++] = 0x82;
273 buffer[pos++] = 0;
274 buffer[pos++] = 0;
275
276 /* Encode each ASDU */
277 for (int i = 0; i < asduCount; i++) {
278 buffer[pos++] = 0x30; /* SEQUENCE */
279 size_t asduLenPos = pos;
280 buffer[pos++] = 0;
281
282 /* svID */
283 size_t svIDLen = strlen(g_config.svID);
284 if (svIDLen == 0) {
285 buffer[pos++] = 0x80;
286 buffer[pos++] = 0x04;
287 memcpy(buffer + pos, "MU01", 4); pos += 4;
288 } else {
289 buffer[pos++] = 0x80;
290 buffer[pos++] = (uint8_t)svIDLen;
291 memcpy(buffer + pos, g_config.svID, svIDLen); pos += svIDLen;
292 }
293
294 /* smpCnt */
295 uint16_t smpCnt = (uint16_t)(baseSmpCnt + i);
296 buffer[pos++] = 0x82;
297 buffer[pos++] = 0x02;
298 buffer[pos++] = (smpCnt >> 8) & 0xFF;
299 buffer[pos++] = smpCnt & 0xFF;
300
301 /* confRev */
302 buffer[pos++] = 0x83;
303 buffer[pos++] = 0x04;
304 buffer[pos++] = (g_config.confRev >> 24) & 0xFF;
305 buffer[pos++] = (g_config.confRev >> 16) & 0xFF;
306 buffer[pos++] = (g_config.confRev >> 8) & 0xFF;
307 buffer[pos++] = g_config.confRev & 0xFF;
308
309 /* smpSynch */
310 buffer[pos++] = 0x85;
311 buffer[pos++] = 0x01;
312 buffer[pos++] = g_config.smpSynch;
313
314 /* seqData (dynamic channels) */
315 uint8_t channelCount = g_config.channelCount;
316 uint8_t seqDataLen = channelCount * 8;
317 buffer[pos++] = 0x87;
318 buffer[pos++] = seqDataLen;
319
320 const int32_t* samples = samplesArray[i];
321 for (int ch = 0; ch < channelCount; ch++) {
322 int32_t val = samples[ch];
323 buffer[pos++] = (val >> 24) & 0xFF;
324 buffer[pos++] = (val >> 16) & 0xFF;
325 buffer[pos++] = (val >> 8) & 0xFF;
326 buffer[pos++] = val & 0xFF;
327 buffer[pos++] = 0x00;
328 buffer[pos++] = 0x00;
329 buffer[pos++] = 0x00;
330 buffer[pos++] = 0x00;
331 }
332
333 buffer[asduLenPos] = (uint8_t)(pos - asduLenPos - 1);
334 }
335
336 /* Fill seqASDU length (2-byte) */
337 size_t seqASDULen = pos - seqASDULenPos - 3;
338 buffer[seqASDULenPos + 1] = (seqASDULen >> 8) & 0xFF;
339 buffer[seqASDULenPos + 2] = seqASDULen & 0xFF;
340
341 /* Fill savPdu length (2-byte) */
342 size_t savPduLen = pos - savPduLenPos - 3;
343 buffer[savPduLenPos + 1] = (savPduLen >> 8) & 0xFF;
344 buffer[savPduLenPos + 2] = savPduLen & 0xFF;
345
346 /* APDU Length: savPduLen + 12 (matches Node.js formula for multi-ASDU) */
347 size_t apduLen = savPduLen + 12;
348 buffer[lengthPos] = (apduLen >> 8) & 0xFF;
349 buffer[lengthPos + 1] = apduLen & 0xFF;
350
351 /* Pad to minimum frame size */
352 while (pos < 60) buffer[pos++] = 0;
353
354 *outSize = pos;
355 return 0;

362size_t sv_encoder_get_frame_size(void) {
363 std::lock_guard<std::mutex> lock(g_mutex);
364
365 size_t headerSize = 14 + (g_config.vlanID > 0 ? 4 : 0);
366 size_t svHeaderSize = 8;
367 size_t savPduOverhead = 10;
368 size_t svIDLen = strlen(g_config.svID);
369 size_t seqDataSize = 2 + (g_config.channelCount * 8); // 2 bytes TL + channels * 8
370 size_t asduSize = 2 + svIDLen + 4 + 6 + 3 + seqDataSize; /* svID+smpCnt+confRev+smpSynch+seqData */
371
372 size_t total = headerSize + svHeaderSize + savPduOverhead + (asduSize * g_config.asduCount);
373 return total < 60 ? 60 : total;
374}
