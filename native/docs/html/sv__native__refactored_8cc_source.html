109 }
110 delete[] g_frames;

114 delete[] g_frame_lens;
115 g_frame_lens = nullptr;
116 }
117 g_frame_count = 0;
118 g_frame_capacity = 0;

121static bool alloc_frame_cache(int count) {
122 free_frame_cache();
123 g_frames = new (std::nothrow) uint8_t*[count];
124 if (!g_frames) return false;
125 g_frame_lens = new (std::nothrow) size_t[count];
126 if (!g_frame_lens) { delete[] g_frames; g_frames = nullptr; return false; }
127 for (int i = 0; i < count; i++) {
128 g_frames[i] = new (std::nothrow) uint8_t[1500];
129 if (!g_frames[i]) {
130 // Cleanup on failure
131 for (int j = 0; j < i; j++) delete[] g_frames[j];
132 delete[] g_frames; g_frames = nullptr;
133 delete[] g_frame_lens; g_frame_lens = nullptr;
134 return false;
135 }
136 }
137 g_frame_capacity = count;
138 return true;
139}

165static void prebuild_frames() {
166 const double PI = 3.14159265358979323846;
167
168 /* IEC 61850-9-2 §7.2.3: smpCnt must count 0 to (sampleRate-1) per second.
169 * Pre-build one full second of frames so smpCnt covers the complete range.
170 * packets_per_second = sampleRate / asduCount (for single ASDU = sampleRate). */
171 int packets_per_second = (int)g_config.sampleRate;
172 if (g_config.asduCount > 1)
173 packets_per_second = (int)(g_config.sampleRate / g_config.asduCount);
174
175 if (packets_per_second > MAX_PREBUILT_FRAMES) {
176 printf("[publisher] WARNING: packets_per_second %d exceeds max %d, clamping\n",
177 packets_per_second, MAX_PREBUILT_FRAMES);
178 packets_per_second = MAX_PREBUILT_FRAMES;
179 }
180 if (packets_per_second < 1) packets_per_second = 1;
181
182 if (!alloc_frame_cache(packets_per_second)) {
183 printf("[publisher] ERROR: Failed to allocate frame cache for %d frames (%.1f MB)\n",
184 packets_per_second, (double)packets_per_second * 1500.0 / (1024.0 * 1024.0));
185 g_frame_count = 0;
186 return;
187 }
188 g_frame_count = packets_per_second;
189
190 printf("[publisher] Building %d frames (1 sec, smpCnt 0-%d), %d ASDUs, %d channels\n",
191 packets_per_second, packets_per_second - 1,
192 g_config.asduCount, g_config.channelCount);
193
194 /* Configure encoder */
195 SvEncoderConfig enc = {0};
196 strncpy(enc.svID, g_config.svID, sizeof(enc.svID) - 1);
197 enc.appID = g_config.appID;
198 enc.confRev = g_config.confRev;
199 enc.smpSynch = g_config.smpSynch;
200 memcpy(enc.srcMAC, g_config.srcMAC, 6);
201 memcpy(enc.dstMAC, g_config.dstMAC, 6);
202 enc.vlanPriority = g_config.vlanPriority;
203 enc.vlanID = g_config.vlanID;
204 enc.asduCount = g_config.asduCount;
205 enc.channelCount = g_config.channelCount;
206 sv_encoder_set_config(&enc);
207
208 /* Build frames — one per packet for the full second.
209 * smpCnt = i ranges 0 to (packets_per_second - 1), e.g. 0-3999 at 4000 Hz. */
210 for (int i = 0; i < packets_per_second; i++) {
211 double t = (double)i / (double)g_config.sampleRate;
212 int32_t samples[SV_MAX_CHANNELS] = {0}; // Support up to 20 channels
213
214 {
215 std::lock_guard<std::mutex> lock(g_eq_mutex);
216 g_eq_processor.generate9_2LESamples(t, samples, g_config.channelCount);
217 }
218
219 size_t size = 1500;
220 sv_encoder_encode_packet(i, samples, g_frames[i], &size);
221 g_frame_lens[i] = size;
222 }
223
224 printf("[publisher] Frame size: %zu bytes\n", g_frame_lens[0]);
225}

260/*============================================================================
261 * Publisher Loop — Batch Mode (sendqueue)
262 *
263 * Two-tier strategy for maximum throughput with minimal duplicate timestamps:
264 *
265 * TIER 1 — PRECISE (<=4800 pps, interval >=208us)
266 * 1 packet per sendqueue, sync=1. NdisMSleep works at these intervals.
267 *
268 * TIER 2 — HIGH-SPEED (>4800 pps, all rates up to line-rate)
269 * sync=0 blasts each batch at wire speed (no kernel NdisMSleep).
270 * Adaptive batch size targets ~500us of wire time per burst.
271 * Userspace QPC spin-wait between batches maintains target rate
272 * and creates inter-burst gaps for receiver NIC timestamp delineation.
273 * At near-line-rate the wire itself limits throughput; smaller batches
274 * create frequent micro-gaps from queue create/fill overhead (~10-30us)
275 * giving the receiver NIC more chances to fire separate interrupts.
276 * Thread is elevated to TIME_CRITICAL with CPU affinity to minimize jitter.
277 *============================================================================*/
278
279static void publisher_loop_batch() {
280 printf("[publisher] Starting batch mode (sendqueue)\n");
281
282 elevate_thread_priority();
283 prebuild_frames();
284
285 if (g_frame_count == 0) {
286 printf("[publisher] ERROR: No frames built, aborting\n");
287 restore_thread_priority();
288 g_running.store(false);
289 return;
290 }
291
292 npcap_stats_reset();
293 npcap_stats_session_start();
294
295 size_t frame_len = g_frame_lens[0];
296 uint64_t total_packets = 0;
297
298 /* Calculate timing parameters */
299 uint64_t packets_per_sec = g_config.sampleRate / g_config.asduCount;
300 double interval_us = (g_config.asduCount * 1000000.0) / (double)g_config.sampleRate;
301 double expected_mbps = (double)packets_per_sec * frame_len * 8.0 / 1e6;
302
303 /* Wire time per frame at 1 Gbps (frame + preamble/SFD/IFG = +20 bytes) */
304 double wire_time_us = (double)(frame_len + 20) * 8.0 / 1000.0;
305
306 printf("[publisher] Target: %llu pps, %.2f Mbps, interval %.2f us, wire %.2f us\n",
307 (unsigned long long)packets_per_sec, expected_mbps, interval_us, wire_time_us);
308
309 /* Persistent frame index across batches so smpCnt cycles through all frames */
310 uint64_t frame_idx = 0;
311
312 /*----------------------------------------------------------------------
313 * TIER 1 — PRECISE: ≤4800 pps (standard IEC 61850-9-2 rate)
314 * 1 pkt/queue, sync=1. NdisMSleep precision is adequate because
315 * the interval (≥208µs) is well above its ~15µs granularity.
316 *----------------------------------------------------------------------*/
317 if (packets_per_sec <= 4800) {
318 printf("[publisher] Mode: PRECISE (1 pkt/queue, sync=1)\n");
319 unsigned int queue_size = (unsigned int)(1 * (frame_len + 24) + 4096);
320 uint64_t ts_accum = 0;
321
322 while (g_running.load() && !check_duration_elapsed()) {
323 void* queue = npcap_queue_create(queue_size);
324 if (!queue) { std::this_thread::sleep_for(std::chrono::milliseconds(1)); continue; }
325
326 npcap_queue_add(queue, g_frames[frame_idx % g_frame_count], frame_len, ts_accum);
327 frame_idx++;
328 ts_accum += (uint64_t)interval_us;
329
330 unsigned int sent = npcap_queue_transmit(queue, 1);
331 if (sent > 0) {
332 total_packets++;
333 npcap_stats_record_packet(frame_len);
334 }
335 npcap_queue_destroy(queue);
336 }
337 }
338 /*----------------------------------------------------------------------
339 * TIER 2 — HIGH-SPEED: >4800 pps (all rates up to line-rate)
340 *
341 * sync=0 blasts each batch at wire speed (no NdisMSleep).
342 * Adaptive batch sizing targets ~500us of wire time:
343 * - Amortizes per-queue syscall overhead
344 * - Creates ~10-30us natural gaps from queue create/fill overhead
345 * - Gives receiver NIC time to raise interrupts between bursts
346 *
347 * Userspace QPC spin-wait between batches maintains target rate.
348 * At near-line-rate (wire_util > 90%), spin-wait exits instantly
349 * and the wire itself is the rate limiter. Throughput stays at
350 * maximum while the frequent micro-gaps from smaller batches
351 * minimize duplicate timestamps at the receiver.
352 *----------------------------------------------------------------------*/
353 else {
354 /* Adaptive batch: target ~500us of wire time per burst.
355 * Small enough for frequent micro-gaps, large enough to
356 * amortize syscall overhead. */
357 uint64_t batch = (uint64_t)(500.0 / wire_time_us);
358 if (batch < 50) batch = 50;
359 if (batch > 5000) batch = 5000;
360
361 double batch_wire_us = batch * wire_time_us;
362 double batch_target_us = batch * interval_us;
363 bool wire_limited = (batch_target_us < batch_wire_us * 1.05);
364
365 auto batch_duration = std::chrono::nanoseconds((int64_t)(batch_target_us * 1000.0));
366
367 printf("[publisher] Mode: HIGH-SPEED (batch=%llu, sync=0%s)\n",
368 (unsigned long long)batch,
369 wire_limited ? ", wire-speed limited" : " + QPC pacing");
370 printf("[publisher] batch_wire: %.0f us, target: %.0f us, util: %.0f%%\n",
371 batch_wire_us, batch_target_us, expected_mbps / 10.0);
372
373 if (wire_limited) {
374 printf("[publisher] NOTE: Target %.0f Mbps saturates the link.\n", expected_mbps);
375 printf("[publisher] Micro-gaps from queue fill overhead (~10-30us)\n");
376 printf("[publisher] provide timestamp breaks at the receiver.\n");
377 }
378
379 unsigned int queue_size = (unsigned int)(batch * (frame_len + 24) + 4096);
380 auto next_batch = std::chrono::high_resolution_clock::now();
381 uint64_t batch_count = 0;
382
383 while (g_running.load() && !check_duration_elapsed()) {
384 void* queue = npcap_queue_create(queue_size);
385 if (!queue) { std::this_thread::sleep_for(std::chrono::milliseconds(1)); continue; }
386
387 for (uint64_t i = 0; i < batch; i++) {
388 npcap_queue_add(queue, g_frames[frame_idx % g_frame_count], frame_len, 0);
389 frame_idx++;
390 }
391
392 /* sync=0: blast batch at wire speed — no kernel pacing */
393 unsigned int sent = npcap_queue_transmit(queue, 0);
394 if (sent > 0) {
395 uint32_t pkts = sent / (uint32_t)frame_len;
396 total_packets += pkts;
397 for (uint32_t i = 0; i < pkts; i++) npcap_stats_record_packet(frame_len);
398 }
399 npcap_queue_destroy(queue);
400
401 /* QPC spin-wait: paces batches when target < wire speed.
402 * At wire-saturated rates, next_batch falls behind so the
403 * spin exits instantly — zero overhead, max throughput.
404 * _mm_pause() reduces CPU power while spinning. */
405 next_batch += batch_duration;
406 while (std::chrono::high_resolution_clock::now() < next_batch) {
407 spin_pause();
408 }
409
410 /* Re-anchor every 200 batches to prevent cumulative drift */
411 batch_count++;
412 if ((batch_count % 200) == 0) {
413 next_batch = std::chrono::high_resolution_clock::now();
414 }
415 }
416 }
417
418 restore_thread_priority();
419
420 /* Handle repeat */
421 if (check_duration_elapsed() && g_config.repeatEnabled) {
422 g_repeat_cycle.fetch_add(1);
423 uint32_t cycle = g_repeat_cycle.load();
424
425 if (g_config.repeatInfinite || cycle < g_config.repeatCount) {
426 printf("[publisher] Repeat cycle %u\n", cycle + 1);
427 g_start_time_ms.store(get_time_ms());
428 g_duration_complete.store(false);
429 publisher_loop_batch();
430 return;
431 }
432 }
433
434 npcap_stats_session_end();
435
436 if (check_duration_elapsed()) {
437 g_duration_complete.store(true);
438 g_running.store(false);
439 }
440
441 printf("[publisher] Complete: %llu packets\n", (unsigned long long)total_packets);
442}

444/*============================================================================
445 * Publisher Loop — Send Immediate (USB Adapter Friendly)
446 *
447 * Uses pcap_sendpacket for each individual packet.
448 * Optimized for USB Ethernet adapters:
449 * - pcap_set_immediate_mode(1) disables driver buffering (in npcap_open)
450 * - TIME_CRITICAL thread prevents OS preemption during inter-packet gaps
451 * - timeBeginPeriod(1) gives 1ms sleep resolution for hybrid pacing
452 * - CPU affinity pins thread to avoid cross-core migration jitter
453 * - _mm_pause() spin-wait for sub-us precision without wasting power
454 *
455 * Pacing modes:
456 * - Low rates (<=4800 pps): hybrid sleep + _mm_pause spin
457 * - Med rates (<=50000 pps): pure _mm_pause spin loop
458 * - High rates (>50000 pps): no pacing, max throughput
459 *
460 * At 4800 Hz (208us interval) with USB 2.0 (125us microframe):
461 * Each packet lands in a separate USB microframe -> distinct timestamps.
462 *============================================================================*/
463
464static void publisher_loop_immediate() {
465 printf("[publisher] Starting SEND-IMMEDIATE mode (pcap_sendpacket per packet)\n");
466
467 elevate_thread_priority();
468
469 prebuild_frames();
470 if (g_frame_count == 0) {
471 printf("[publisher] ERROR: No frames built, aborting\n");
472 restore_thread_priority();
473 g_running.store(false);
474 return;
475 }
476
477 npcap_stats_reset();
478 npcap_stats_session_start();
479
480 uint64_t total_packets = 0;
481 uint64_t total_failures = 0;
482
483 /* Calculate timing parameters */
484 uint64_t packets_per_sec = g_config.sampleRate / g_config.asduCount;
485 double interval_us = (g_config.asduCount * 1000000.0) / (double)g_config.sampleRate;
486 double expected_mbps = (double)packets_per_sec * g_frame_lens[0] * 8.0 / 1e6;
487
488 /* Choose pacing strategy */
489 enum PacingMode { PACING_SLEEP, PACING_SPIN, PACING_NONE };
490 PacingMode pacing;
491
492 bool usbMode = (g_config.sendMode == SEND_MODE_USB);
493 int usbPadSize = g_config.usbPadSize;
494 int usbGapUs = (g_config.usbMinGapUs > 0) ? g_config.usbMinGapUs : 130;
495
496 if (usbMode) {
497 pacing = PACING_SPIN;
498 printf("[publisher] Pacing: USB-OPTIMIZED (forced SPIN + %d us gap)\n", usbGapUs);
499 if (usbPadSize > 0)
500 printf("[publisher] USB frame padding: %d bytes\n", usbPadSize);
501 } else if (packets_per_sec <= 4800) {
502 pacing = PACING_SLEEP;
503 printf("[publisher] Pacing: SLEEP+SPIN hybrid (low rate, USB friendly)\n");
504 } else if (packets_per_sec <= 50000) {
505 pacing = PACING_SPIN;
506 printf("[publisher] Pacing: SPIN _mm_pause (medium rate, high precision)\n");
507 } else {
508 pacing = PACING_NONE;
509 printf("[publisher] Pacing: NONE (max throughput)\n");
510 }
511
512 printf("[publisher] Target: %llu pps, %.2f Mbps, interval %.2f us\n",
513 (unsigned long long)packets_per_sec, expected_mbps, interval_us);
514
515 /* Persistent frame index cycles through pre-built cache */
516 uint64_t frame_idx = 0;
517 auto interval_dur = std::chrono::nanoseconds((int64_t)(interval_us * 1000.0));
518
519 /* Epoch-based absolute scheduling (industry standard).
520 *
521 * target[N] = epoch + N * interval
522 *
523 * This ensures the average packet rate is EXACTLY correct regardless
524 * of how long each pcap_sendpacket() call takes.
525 *
526 * If pcap_sendpacket takes longer than the interval, we fall behind
527 * schedule. To handle this:
528 * - Small lag: send next packet immediately (natural catch-up)
529 * - Large lag (>3 intervals behind): re-anchor epoch to prevent
530 * unbounded bursts of back-to-back packets
531 *
532 * This is the same approach used by tcpreplay, Linux pktgen, and
533 * professional traffic generators (DPDK TRex, MoonGen, etc.).
534 */
535 auto epoch = std::chrono::high_resolution_clock::now();
536 uint64_t pkt_num = 0;
537
538 while (g_running.load() && !check_duration_elapsed()) {
539 /* Send packet immediately */
540 int idx = (int)(frame_idx % g_frame_count);
541 int result;
542 if (usbMode && usbPadSize > 0 && usbPadSize > (int)g_frame_lens[idx]) {
543 result = npcap_send_packet_padded(g_frames[idx], g_frame_lens[idx], (size_t)usbPadSize);
544 } else {
545 result = npcap_send_packet(g_frames[idx], g_frame_lens[idx]);
546 }
547
548 if (result == 0) {
549 total_packets++;
550 npcap_stats_record_packet(g_frame_lens[idx]);
551 } else {
552 total_failures++;
553 npcap_stats_record_failure();
554 }
555
556 frame_idx++;
557 pkt_num++;
558
559 if (pacing == PACING_NONE)
560 continue;
561
562 /* Absolute target for the NEXT packet */
563 auto target = epoch + pkt_num * interval_dur;
564 auto now = std::chrono::high_resolution_clock::now();
565
566 if (target <= now) {
567 /* Behind schedule — check how far */
568 auto behind = std::chrono::duration_cast<
569 std::chrono::nanoseconds>(now - target);
570 if (behind > interval_dur) {
571 /* Behind by >1 interval — re-anchor to prevent burst.
572 * Allows at most 1 catch-up packet (small lag < 1 interval)
573 * but prevents back-to-back bursts of 3-4 packets that
574 * appear as 0.000000 deltas in Wireshark. */
575 epoch = now;
576 pkt_num = 0;
577 }
578
579 /* USB mode: even when behind, enforce minimum gap between
580 * sends so the USB host controller doesn't batch packets
581 * into the same microframe (125µs). Configurable gap. */
582 if (usbMode) {
583 auto minTarget = now + std::chrono::microseconds(usbGapUs);
584 while (std::chrono::high_resolution_clock::now() < minTarget)
585 spin_pause();
586 }
587 continue;
588 }
589
590 /* Ahead of schedule — pace to exact target */
591 switch (pacing) {
592 case PACING_SLEEP: {
593 auto remaining = std::chrono::duration_cast<
594 std::chrono::microseconds>(target - now);
595 if (remaining.count() > 80) {
596 std::this_thread::sleep_for(
597 remaining - std::chrono::microseconds(80));
598 }
599 while (std::chrono::high_resolution_clock::now() < target) {
600 spin_pause();
601 }
602 break;
603 }
604 case PACING_SPIN:
605 while (std::chrono::high_resolution_clock::now() < target) {
606 spin_pause();
607 }
608 break;
609 default:
610 break;
611 }
612 }
613
614 restore_thread_priority();
615
616 /* Handle repeat */
617 if (check_duration_elapsed() && g_config.repeatEnabled) {
618 g_repeat_cycle.fetch_add(1);
619 uint32_t cycle = g_repeat_cycle.load();
620
621 if (g_config.repeatInfinite || cycle < g_config.repeatCount) {
622 printf("[publisher] Repeat cycle %u\n", cycle + 1);
623 g_start_time_ms.store(get_time_ms());
624 g_duration_complete.store(false);
625 publisher_loop_immediate();
626 return;
627 }
628 }
629
630 npcap_stats_session_end();
631
632 if (check_duration_elapsed()) {
633 g_duration_complete.store(true);
634 g_running.store(false);
635 }

641static void publisher_loop_single() {
642 printf("[publisher] Starting single-packet mode\n");
643
644 npcap_stats_reset();
645 npcap_stats_session_start();
646
647 SvEncoderConfig enc = {0};
648 strncpy(enc.svID, g_config.svID, sizeof(enc.svID) - 1);
649 enc.appID = g_config.appID;
650 enc.confRev = g_config.confRev;
651 enc.smpSynch = g_config.smpSynch;
652 memcpy(enc.srcMAC, g_config.srcMAC, 6);
653 memcpy(enc.dstMAC, g_config.dstMAC, 6);
654 enc.vlanPriority = g_config.vlanPriority;
655 enc.vlanID = g_config.vlanID;
656 enc.asduCount = 1;
657 enc.channelCount = g_config.channelCount;
658 sv_encoder_set_config(&enc);
659
660 uint32_t smpCnt = 0;
661 int samples_per_cycle = (int)(g_config.sampleRate / g_config.frequency);
662 if (samples_per_cycle < 1) samples_per_cycle = 1;
663 double interval_us = 1000000.0 / g_config.sampleRate;
664 auto lastTime = std::chrono::high_resolution_clock::now();
665 uint8_t packet[1500];
666
667 while (g_running.load() && !check_duration_elapsed()) {
668 double t = (double)smpCnt / (double)g_config.sampleRate;
669 int32_t samples[SV_MAX_CHANNELS] = {0}; // Support up to 20 channels
670
671 {
672 std::lock_guard<std::mutex> lock(g_eq_mutex);
673 g_eq_processor.generate9_2LESamples(t, samples, g_config.channelCount);
674 }
675
676 size_t size = sizeof(packet);
677 sv_encoder_encode_packet(smpCnt % (uint32_t)g_config.sampleRate, samples, packet, &size);
678
679 if (npcap_send_packet(packet, size) == 0) {
680 npcap_stats_record_packet(size);
681 } else {
682 npcap_stats_record_failure();
683 }
684
685 smpCnt++;
686
687 auto target = lastTime + std::chrono::microseconds((int64_t)interval_us);
688 while (std::chrono::high_resolution_clock::now() < target) std::this_thread::yield();
689 lastTime = target;
690 }
691
692 npcap_stats_session_end();
693
694 if (check_duration_elapsed()) {
695 g_duration_complete.store(true);
696 g_running.store(false);
697 }
698}

753int npcap_publisher_start(void) {
754 if (!npcap_is_open()) {
755 snprintf(g_error_buffer, sizeof(g_error_buffer), "No interface open");
756 return -1;
757 }
758
759 if (g_running.load()) return 0;
760
761 if (sv_mp_is_running()) {
762 snprintf(g_error_buffer, sizeof(g_error_buffer),
763 "Multi-publisher is already running. Stop it before using single-publisher.");
764 return -1;
765 }
766
767 g_start_time_ms.store(get_time_ms());
768 g_duration_complete.store(false);
769 g_running.store(true);
770
771 /* Dispatch based on configured send mode */
772 switch (g_config.sendMode) {
773 case SEND_MODE_QUEUE:
774 /* Force SendQueue (batch) mode */
775 if (npcap_sendqueue_available()) {
776 printf("[publisher] Mode: FORCED SendQueue (batch)\n");
777 g_thread = std::thread(publisher_loop_batch);
778 } else {
779 printf("[publisher] WARNING: SendQueue requested but not available, falling back to single\n");
780 g_thread = std::thread(publisher_loop_single);
781 }
782 break;
783
784 case SEND_MODE_PACKET:
785 /* Force SendPacket (immediate) mode */
786 printf("[publisher] Mode: FORCED SendPacket (immediate)\n");
787 g_thread = std::thread(publisher_loop_immediate);
788 break;
789
790 case SEND_MODE_USB:
791 /* USB-Optimized: immediate + forced SPIN + minimum gap */
792 printf("[publisher] Mode: USB-OPTIMIZED (spin pacing + min gap)\n");
793 g_thread = std::thread(publisher_loop_immediate);
794 break;
795
796 case SEND_MODE_AUTO:
797 default:
798 /* Auto-detect: use SendQueue if available, else fall back to single */
799 if (npcap_sendqueue_available()) {
800 printf("[publisher] Mode: AUTO -> SendQueue (batch)\n");
801 g_thread = std::thread(publisher_loop_batch);
802 } else {
803 printf("[publisher] Mode: AUTO -> single (no sendqueue)\n");
804 g_thread = std::thread(publisher_loop_single);
805 }
806 break;
807 }
808
809 return 0;
810}
